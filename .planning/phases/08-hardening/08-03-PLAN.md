```typescript
/**
 * Format an ISO timestamp (e.g., "2026-03-06T12:00:00.000Z") to "DD/MM/YYYY".
 * Returns '-' for null/undefined.
 */
export function formatDate(iso: string | null): string {
  if (!iso) return '-';
  return new Date(iso).toLocaleDateString('es-AR', {
    day: '2-digit',
    month: '2-digit',
    year: 'numeric',
  });
}

/**
 * Format a date-only string (e.g., "2026-03-06") to "DD/MM/YYYY".
 * Avoids Date constructor to prevent timezone shift on date-only strings.
 */
export function formatDateDisplay(dateStr: string): string {
  const [year, month, day] = dateStr.split('-');
  return `${day}/${month}/${year}`;
}
```
