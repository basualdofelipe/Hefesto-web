import { auth } from '@/auth';

export const proxy = auth((req) => {
  const { pathname } = req.nextUrl;
  // Unauthenticated: redirect to /login (except public pages)
  if (!req.auth && pathname !== '/login' && pathname !== '/acceso-denegado') {
    const loginUrl = new URL('/login', req.nextUrl.origin);
    loginUrl.searchParams.set('callbackUrl', pathname);
    return Response.redirect(loginUrl);
  }
  // Auth'd but no backend token: redirect to /acceso-denegado
  if (req.auth && !req.auth.accessToken && pathname !== '/acceso-denegado' && pathname !== '/login') {
    return Response.redirect(new URL('/acceso-denegado', req.nextUrl.origin));
  }
});

export const config = {
  matcher: ['/((?!api|_next/static|_next/image|favicon.ico|fonts|images).*)'],

**apiClientFetch 401/403 handling:**

Update `nemea-front/src/lib/api-client.ts` to differentiate 401 (session expired) from 403 (forbidden) from other errors:

```typescript
'use client';

import { signOut } from 'next-auth/react';

const API_URL = process.env.NEXT_PUBLIC_API_URL;

export async function apiClientFetch<T>(
  path: string,
  token: string,
  options: RequestInit = {},
): Promise<T> {
  const res = await fetch(`${API_URL}${path}`, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${token}`,
      ...options.headers,
    },
  });

  if (res.status === 401) {
    // JWT expired or invalidated -- redirect to login
    // Use signOut to clear the NextAuth session cookie cleanly
    void signOut({ callbackUrl: '/login' });
    throw new Error('Sesion expirada. Redirigiendo al login...');
  }

  if (res.status === 403) {
    throw new Error('No tienes permisos para realizar esta accion');
  }

  if (!res.ok) {
    const error = await res
      .json()
      .catch(() => ({ message: `Error ${res.status}` }));
    throw new Error(error.message ?? `API error: ${res.status}`);
  }

  return res.json() as Promise<T>;
}
```

Key design decisions:
- Use `signOut()` from next-auth/react instead of `window.location.href` to cleanly clear the NextAuth session cookie before redirecting. This prevents a loop where an expired session keeps trying to use the old token.
- 403 throws a user-friendly Spanish error message that will surface in the toast (all callers use `toast.error(error.message)`)
- The `void signOut()` is fire-and-forget -- the thrown error prevents the caller from continuing
- Import `signOut` at top level. Since this file is already `'use client'`, this is fine.
